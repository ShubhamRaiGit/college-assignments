CREATE VIEW IF NOT EXISTS stackexchange_view as
select cast(Id as int) as Id,
cast(PostTypeId as tinyint) as PostTypeId,
AcceptedAnswerId ,
ParentId ,
CreationDate ,
DeletionDate ,
cast(Score as int) as Score,
cast(ViewCount as int) as ViewCount,
Body,
cast(OwnerUserId as int) as OwnerUserId,
OwnerDisplayName as DisplayName,
LastEditorUserId,
LastEditorDisplayName ,
LastEditDate ,
LastActivityDate ,
Title ,
Tags ,
AnswerCount ,
CommentCount ,
FavoriteCount ,
ClosedDate ,
CommunityOwnedDate ,
ContentLicense from stackexchangeposts;
